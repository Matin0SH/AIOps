!
! MANAGEMENT Switch Configuration
! Role: Core L2 Switch, STP Root Bridge
!
hostname MANAGEMENT
service password-encryption
!
enable secret cisco
username admin privilege 15 secret cisco
!
! VLAN configuration
vlan 10
 name MANAGEMENT
!
! Management SVI for in-band access
interface Vlan10
 description *** Management SVI ***
 ip address 10.10.10.10 255.255.255.0
 no shutdown
!
ip default-gateway 10.10.10.1
!
! STP configuration - make this the root bridge
spanning-tree mode rapid-pvst
spanning-tree vlan 10 priority 24576
!
! Trunk to EDGE-R1 (router-on-a-stick)
interface GigabitEthernet0/1
 description *** Trunk to EDGE-R1 ***
 switchport trunk encapsulation dot1q
 switchport mode trunk
 switchport trunk allowed vlan 10
 no shutdown
!
! Trunk to CORE-SW1
interface GigabitEthernet0/2
 description *** Trunk to CORE-SW1 ***
 switchport trunk encapsulation dot1q
 switchport mode trunk
 switchport trunk allowed vlan 10
 no shutdown
!
! Trunk to CORE-SW2
interface GigabitEthernet0/3
 description *** Trunk to CORE-SW2 ***
 switchport trunk encapsulation dot1q
 switchport mode trunk
 switchport trunk allowed vlan 10
 no shutdown
!
! Enable CDP
cdp run
!
! VTY access
line vty 0 4
 login local
 transport input ssh
!
! Generate RSA keys for SSH
ip ssh version 2
!
end
wr mem
!
