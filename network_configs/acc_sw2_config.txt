!
! ACC-SW2 Configuration
! Role: Access Layer L2 Switch
!
hostname ACC-SW2
service password-encryption
!
enable secret cisco
username admin privilege 15 secret cisco
!
ip domain-name aiops.local
!
! VLAN configuration
vlan 10
 name MANAGEMENT
!
! Management SVI
interface Vlan10
 description *** Management SVI ***
 ip address 10.10.10.5 255.255.255.0
 no shutdown
!
! Default gateway
ip default-gateway 10.10.10.1
!
! STP configuration - access layer (default priority)
spanning-tree mode rapid-pvst
spanning-tree vlan 10 priority 32768
!
! Trunk to CORE-SW1 (Cross-Connect)
interface GigabitEthernet0/1
 description *** Trunk to CORE-SW1 (Cross-Connect) ***
 switchport trunk encapsulation dot1q
 switchport mode trunk
 switchport trunk allowed vlan 10
 no shutdown
!
! Trunk to CORE-SW2
interface GigabitEthernet0/2
 description *** Trunk to CORE-SW2 ***
 switchport trunk encapsulation dot1q
 switchport mode trunk
 switchport trunk allowed vlan 10
 no shutdown
!
! Access ports for end devices
interface GigabitEthernet0/3
 description *** Access Port ***
 switchport mode access
 switchport access vlan 10
 spanning-tree portfast
 spanning-tree bpduguard enable
 shutdown
!
! Enable CDP
cdp run
!
! VTY access
line vty 0 4
 login local
 transport input ssh
!
! Generate RSA keys for SSH
ip ssh version 2
!
end
wr mem
!
